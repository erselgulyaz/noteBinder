EXAMPLE 1

Exiting in the input element is called I prepared a sample of noteBinder message. In writing or if there is no will change the messages.

					$(".form-input").blur(function(){

						if ($(this).val().length > 0) {

							var options = {

								openEffect: "slideToVer",

								closeEffect: "slideToHor",

								maxItem: 5,

								message: "<div class='success'><a class='ico icon-cancel noteBinder-close'></a><div class='ico icon-ok'></div><span>input have a text :)</span></div>",

								verticalPos: "bottom",

								horizontalPos: "right",

							};

						}else{

							var options = {

								openEffect: "bounce",

								closeEffect: "fade",

								maxItem: 3,

								waiting: 7000,

								message: "<div class='error'><a class='ico icon-cancel noteBinder-close'></a><div class='ico icon-attention-alt'></div><span>input is empty :(</span></div>",

								posForGroup: "after"

							};

						}

						noteBinder.init(options);

					});
