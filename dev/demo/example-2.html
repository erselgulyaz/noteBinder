EXAMPLE 2

Adding an radio buttons as a survey. You will guess multiple radio buttons and one true. We are calling noteBinder message true or false answers.

					$(".radio-button").change(function(){

						if ($(this).attr("id") == "two") {

							var options = {

								openEffect: "fade",

								closeEffect: "slideToVer",

								maxItem: 5,

								message: "<div class='success'><a class='ico icon-cancel noteBinder-close'></a><div class='ico icon-ok'></div><span>your selection is true :)</span></div>",

								verticalPos: "bottom",

								horizontalPos: "right",

							};

						}else{

							var options = {

								openEffect: "bounce",

								closeEffect: "slideToHor",

								maxItem: 3,

								waiting: 4000,

								message: "<div class='error'><a class='ico icon-cancel noteBinder-close'></a><div class='ico icon-attention-alt'></div><span>your selection is false :(</span></div>",

								verticalPos: "bottom",

								horizontalPos: "right",

							};

						}

						noteBinder.init(options);

					});
