EXAMPLE 3

Adding an checkbox elements. we will writing callback functions of Element on check. noteBinder message pushing and options added checkbox title.

					$(".check-button").change(function(){

						var itemLabel = $(this).next("label").text();

						if ($(this).is(":checked")) {

							var options = {

								openEffect: "slideToVer",

								closeEffect: "fade",

								message: "<div class='success'><a class='ico icon-cancel noteBinder-close'></a><div class='ico icon-ok'></div><span>Your checked item is '"+ itemLabel +"'</span></div>",

								verticalPos: "bottom",

								verticalSpace: 10,

								horizontalPos: "left",

							};

						}else {

							var options = {

								openEffect: "slideToHor",

								closeEffect: "slideToVer",

								maxItem: 3,

								message: "<div class='error'><a class='ico icon-cancel noteBinder-close'></a><div class='ico icon-attention-alt'></div><span>your unchecked item is '"+ itemLabel +"'</span></div>",

								verticalPos: "bottom",

								verticalSpace: 10,

								horizontalPos: "left",

							};

						}

						noteBinder.init(options);

					});
